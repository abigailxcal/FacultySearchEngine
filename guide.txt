
procedure crawlerThread (frontier, num_targets)
    targets_found = 0   
    while not frontier.done() do
        url <— frontier.nextURL()   # get next url
        html <— retrieveURL(url)    # get html of the url
        storePage(url, html)        # store data
            if target_page (html) #if URL is a target page...
                targets_found = targets_found + 1
            if targets_found = num_targets  # all targets have been found
                clear_frontier()
            else
                for each not visited url in parse (html) do
                    frontier.addURL(url)
                end for
    end while
end procedure


main:
    frontier = crawler()
    

for each frontaierURL, extract the HTML and store the url and HTML 
if the html indicates it is a target, we check to see if all the targets have been found yet.
If not, we parse the HTML for additional URLs and add that to our list of URLS to visit

    4. if url has not been visited yet, url is added to the request queue (frontier)
    5. Cont until no noew URLs
